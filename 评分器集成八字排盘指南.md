# 评分器集成八字排盘模块指南

## 概述

本指南说明如何在评分器中集成八字排盘模块，以获取更详细的八字信息，增强评分的准确性和全面性。

## 集成方案

### 1. 导入八字排盘模块

在评分器模块中导入八字排盘模块：

```python
from .八字排盘 import BaZiPanPan, create_bazi_panpan
```

### 2. 在评分方法中集成八字排盘

修改评分器的 `score` 方法，在处理事主信息时创建八字排盘对象：

```python
def score(self, sizhu, event_type, owners=None, house_type=None, 
          shan_xiang=None, zaoxiang=None, zaowei=None, chuangwei=None):
    """
    计算评分（集成八字排盘）
    """
    # 1. 如果提供了事主信息，创建八字排盘对象
    bazi_info = None
    if owners and len(owners) > 0:
        try:
            owner = owners[0]  # 取第一个事主
            if 'birth_year' in owner and 'birth_month' in owner and 'birth_day' in owner:
                # 创建八字排盘对象
                bazi = BaZiPanPan(
                    year=owner['birth_year'],
                    month=owner['birth_month'],
                    day=owner['birth_day'],
                    hour=owner.get('birth_hour', 12),
                    minute=owner.get('birth_minute', 0),
                    gender=owner.get('gender', '男')
                )
                bazi_info = bazi.get_panpan_result()
        except Exception as e:
            print(f"八字排盘失败: {e}")
    
    # 2. 执行原有的评分逻辑
    # ...
    
    # 3. 使用八字信息增强评分
    if bazi_info:
        score_adjustment = self._calculate_bazi_score_adjustment(bazi_info, sizhu)
        self.final_score += score_adjustment
    
    return {
        'score': self.final_score,
        'level': self.level,
        'bazi_info': bazi_info,
        # ... 其他信息
    }
```

### 3. 添加八字评分调整方法

在评分器中添加基于八字信息的评分调整方法：

```python
def _calculate_bazi_score_adjustment(self, bazi_info, day_sizhu):
    """
    根据八字信息计算评分调整
    
    Args:
        bazi_info: 事主八字信息
        day_sizhu: 日课四柱信息
        
    Returns:
        int: 评分调整值
    """
    adjustment = 0
    
    # 1. 日主与日课日主的五行关系
    owner_day_gan = bazi_info['基本信息']['四柱']['日柱'][0]
    day_day_gan = day_sizhu.get('day_gan', '')
    
    if owner_day_gan and day_day_gan:
        # 检查五行生克关系
        from .八字工具整合 import GAN_WUXING, WUXING_SHENG, WUXING_KE
        
        owner_wx = GAN_WUXING.get(owner_day_gan, '')
        day_wx = GAN_WUXING.get(day_day_gan, '')
        
        if owner_wx and day_wx:
            # 日课生日主：加分
            if WUXING_SHENG.get(day_wx) == owner_wx:
                adjustment += 10
            # 日课克日主：减分
            elif WUXING_KE.get(day_wx) == owner_wx:
                adjustment -= 15
            # 日主生日课：加分
            elif WUXING_SHENG.get(owner_wx) == day_wx:
                adjustment += 5
    
    # 2. 日主与日课地支的关系
    owner_day_zhi = bazi_info['基本信息']['四柱']['日柱'][1]
    day_day_zhi = day_sizhu.get('day_zhi', '')
    
    if owner_day_zhi and day_day_zhi:
        from .八字工具整合 import check_liuhe, check_liuchong
        
        # 合：加分
        if check_liuhe(owner_day_zhi, day_day_zhi):
            adjustment += 8
        # 冲：减分
        if check_liuchong(owner_day_zhi, day_day_zhi):
            adjustment -= 12
    
    # 3. 纳音匹配度
    owner_nayin = bazi_info['纳音']['日柱']
    day_nayin = day_sizhu.get('nayin', {}).get('日柱', '')
    
    if owner_nayin and day_nayin and owner_nayin != '未知' and day_nayin != '未知':
        # 提取纳音五行
        nayin_wx_map = {
            '金': ['海中金', '剑锋金', '白蜡金', '砂中金', '金箔金', '钗钏金'],
            '木': ['大林木', '杨柳木', '松柏木', '平地木', '桑柘木', '大溪水', '石榴木'],
            '水': ['涧下水', '泉中水', '长流水', '天河水', '大溪水', '大海水'],
            '火': ['炉中火', '山头火', '霹雳火', '山下火', '覆灯火', '天上火'],
            '土': ['路旁土', '城头土', '屋上土', '壁上土', '沙中土', '大驿土']
        }
        
        owner_wx = None
        day_wx = None
        
        for wx, names in nayin_wx_map.items():
            if owner_nayin in names:
                owner_wx = wx
            if day_nayin in names:
                day_wx = wx
        
        if owner_wx and day_wx:
            # 纳音相同：加分
            if owner_wx == day_wx:
                adjustment += 6
            # 纳音相生：加分
            elif WUXING_SHENG.get(owner_wx) == day_wx:
                adjustment += 4
    
    return adjustment
```

### 4. 添加十二长生评分方法

```python
def _calculate_zhangsheng_score(self, bazi_info):
    """
    计算日主十二长生状态的评分
    
    Args:
        bazi_info: 事主八字信息
        
    Returns:
        int: 十二长生评分
    """
    score = 0
    zhangsheng = bazi_info.get('十二长生', {})
    
    # 检查日支的长生状态
    day_zhi_zhangsheng = zhangsheng.get('日支', '')
    
    # 帝旺、临官、冠带、长生：加分
    if day_zhi_zhangsheng in ['帝旺', '临官', '冠带', '长生']:
        score += 8
    # 衰、病、死、墓、绝：减分
    elif day_zhi_zhangsheng in ['衰', '病', '死', '墓', '绝']:
        score -= 6
    
    return score
```

### 5. 在五行审核中集成

修改 `_check_wu_xing` 方法，集成八字排盘信息：

```python
def _check_wu_xing(self, sizhu, event_type, owners, house_type, shan_xiang,
                  zaoxiang, zaowei, chuangwei):
    """
    正五行审核（集成八字排盘）
    """
    # 原有的神煞和宜忌检查
    shensha_checker = get_checker(event_type)
    self.shensha_list = shensha_checker.check(sizhu, owners)
    
    rule_checker = get_rule_checker(event_type)
    self.yi_list, self.ji_list = rule_checker.check(
        sizhu, owners, house_type, shan_xiang, zaoxiang, zaowei, chuangwei
    )
    
    # 计算五行评分
    wu_xing_score = self.base_score
    for shensha in self.shensha_list:
        wu_xing_score += shensha['score']
    for yi in self.yi_list:
        wu_xing_score += 10
    for ji in self.ji_list:
        wu_xing_score -= 15
    
    # 新增：集成八字排盘模块的功能
    if owners and len(owners) > 0:
        try:
            owner = owners[0]
            if 'birth_year' in owner and 'birth_month' in owner and 'birth_day' in owner:
                # 创建八字排盘对象
                bazi = BaZiPanPan(
                    year=owner['birth_year'],
                    month=owner['birth_month'],
                    day=owner['birth_day'],
                    hour=owner.get('birth_hour', 12),
                    minute=owner.get('birth_minute', 0),
                    gender=owner.get('gender', '男')
                )
                bazi_info = bazi.get_panpan_result()
                
                # 1. 计算日主十二长生状态，影响旺衰评分
                zhangsheng_score = self._calculate_zhangsheng_score(bazi_info)
                wu_xing_score += zhangsheng_score
                
                # 2. 计算八字与日课的匹配度
                bazi_match_score = self._calculate_bazi_score_adjustment(bazi_info, sizhu)
                wu_xing_score += bazi_match_score
                
        except Exception as e:
            print(f"八字排盘集成失败: {e}")
    
    # 判断五行是否合格
    he_ge = wu_xing_score >= 60
    
    # 生成五行评语
    ji_yu = self._generate_wu_xing_jiyu(wu_xing_score, he_ge)
    
    return {
        'he_ge': he_ge,
        'score': wu_xing_score,
        'ji_yu': ji_yu
    }
```

## 集成后的优势

### 1. 更精确的五行分析
- 使用带权重的五行分数计算，更准确地反映五行力量
- 考虑藏干权重，提供更全面的五行分布

### 2. 事主与日课的匹配度
- 分析日主与日课日主的五行生克关系
- 检查地支的冲合关系
- 考虑纳音五行的匹配度

### 3. 日主旺衰分析
- 通过十二长生状态判断日主旺衰
- 根据旺衰状态调整评分

### 4. 更丰富的信息展示
- 显示事主的完整八字信息
- 显示藏干、十神、纳音等详细信息
- 显示大运信息

## 注意事项

### 1. 性能考虑
- 八字排盘计算相对耗时，建议只在需要时创建
- 可以缓存八字排盘结果，避免重复计算

### 2. 错误处理
- 八字排盘可能失败，需要添加异常处理
- 如果八字排盘失败，应该回退到原有评分逻辑

### 3. 向后兼容
- 集成后应该保持向后兼容
- 如果没有提供事主信息，应该使用原有评分逻辑

### 4. 权重调整
- 八字评分调整的权重需要根据实际情况调整
- 建议通过大量测试确定最佳权重
